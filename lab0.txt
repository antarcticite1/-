#include <stdio.h>

int main() {
    int n;
    printf("Input n: ");
    scanf("%d", &n);
    
    int P[n];
    
    
    for (int i = 0; i < n; i++) {
        P[i] = i + 2;
    }
    
    
    for (int i = 0; i < n; i++) {
        int p = P[i];
        
        
        if (p == 0) {
            continue;
        }
        
        
        if (p * p > n) {
            break;
        }
        
        
        for (int j = i + 1; j < n; j++) {
            if (P[j] % p == 0) {
                P[j] = 0;
            }
        }
    }
    
    
    printf("Prostye chisla:\n");
    for (int i = 0; i < n; i++) {
        if (P[i] != 0) {
            printf("%d ", P[i]);
        }
    }
    printf("\n");
    
    return 0;
}
